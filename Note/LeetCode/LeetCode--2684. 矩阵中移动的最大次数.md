func maxMoves(grid [][]int) int {
    n, m := len(grid), len(grid[0])
    ans := 0
    f := make([][]int, n)
    for i := 0; i < n; i ++ {
        f[i] = make([]int, m)
        for j := 0; j < m; j ++ {
            f[i][j] = -1
        }
    }
    for i := 0; i < n; i ++ {
        f[i][0] = 0
    }
    for i := 1; i < m; i ++ {
        for j := 0; j < n; j ++ {
            if grid[j][i] > grid[j][i - 1] && f[j][i - 1] != -1 {
                f[j][i] = f[j][i - 1] + 1
            } else if j + 1 < n && grid[j][i] > grid[j + 1][i - 1] && f[j + 1][i - 1] != -1 {
                f[j][i] = f[j + 1][i - 1] + 1
            } else if j - 1 >= 0 && grid[j][i] > grid[j - 1][i - 1] && f[j - 1][i - 1] != -1 {
                f[j][i] = f[j - 1][i - 1] + 1
            }
            ans = max(ans, f[j][i])
        }
    }
    return ans
}
